Configurations

Property Value Description
title String Title of the table
yAxis 0 to (dataset.length-1) column index of the dataset for Y-axis
xAxis 0 to (dataset.length-1) column index of the dataset for X-axis
pointLabel 0 to (dataset.length-1) column index of the dataset for text labels
width px width of the chart
height px height of the chart
chartType bar Type of the chart
textAngle (-360,+360) Angle between lables and X-axis
barColor color code color of the bars

Sample code for above charts


/******************************************************* Installation of the library ******************************************************/

             <link rel="stylesheet" href="bootstrap.min.css">
             <link rel="stylesheet" href="igviz.css">
             <script src="d3.min.js"> </script>
             <script src="igviz.js"> </script>


/***********************************************************Invoking the Library **********************************************************/
             <script>

                    //dataset used to plot charts
                    var dataTable = {
                    "metadata":{
                        "names":["Country","State",'city',"population"],
                        "types":['C', 'C','C','N']
                      },
                      "data": [

                          ["A","a",6,2],
                        ["A","a",1,2],
                        ["A","b",1,3],
                        ["A","c",3,4],
                        ["B","a",2,1],
                        ["A","b",3,5],
                        ["B","a",1,6],
                        ["B","b",3,5],
                        ["A","c",5,3],
                        ["A", "d",2,9],
                        ["B","e",2,2],
                        ["A","a",1,1],

                      ]
                    }

                    var width = document.getElementById("drill").offsetWidth; //canvas width
                    var height = 270;   //canvas height

                    var config = {
                        "title": "Area By Country",
                        "yAxis": 3,
                        "xAxis": [{"index":0,"child":1,"div":"drill"},{"index":1,"child":2,"div":"drill"}],
                        "pointLabel": 0,
                        "width": width,
                        "height": height,
                        "padding": 60,
                        "chartType": "drill"
                    }



                    //draw a drill downing  chart on div
                    igviz.plot("#drill",config,dataTable);
            </script>
