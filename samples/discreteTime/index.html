IGViz.js Drawing a chart has never been this easy!

Interactive Generic Visualization library for Javascript that is based on d3.js

Configurations

Property Value Description
title String Title of the table
yAxis [0 to (dataset.length-1)] List of variables for Y-axis
xAxis 0 to (dataset.length-1) column index of the dataset for X-axis
width px width of the chart
height px height of the chart
chartType bar Type of the chart
linkMark true | false Link marks required
pointMark true | false Point marks required

Sample code for above charts


/******************************************************* Installation of the library ******************************************************/

             <link rel="stylesheet" href="igviz.css">
             <script src="d3.min.js"> </script>
             <script src="vega.js"> </script>
             <script src="igviz.js"> </script>


/***********************************************************Invoking the Library **********************************************************/
             <script>

                    //dataset used to plot charts
                    var sampleData={
                        "metadata":{
                            "names":["Time","Temperature","humidity"],
                            "types":['T', 'N' ,'N']
                        }    ,
                        "data": [
                        ["October 13, 2014 11:13:00",0,2],

                        ["October 13, 2014 12:14:00",4,0],
                        ["October 14, 2014 1:23:00",2,4],
                        ["October 13, 2014 13:16:00",0,2],
                        ["October 13, 2014 14:15:00",1,3],
                        ["October 14, 2014 5:23:00",3,4],
                        ["October 13, 2014 16:16:00",4,2]


                        ]
                    }




                        var width = document.getElementById("discrete").offsetWidth; //canvas width
                        var height = 270;   //canvas height



                    var config2 = {
                        "title": "Area By Country",
                        "yAxis": [1,2],
                        "xAxis": 0,
                        "width": width,
                        "height": height,
                        "lineMark":true,
                        "pointMark":true,
                        "chartType": "series"

                    }


                    staticChart=igviz.setUp("#discrete",config2,sampleData);
                    staticChart.plot(sampleData.data)



                    </script>
