Configurations

Property Value Description
title String Title of the table
yAxis [0 to (dataset.length-1)] List of column indices for Y-axis variables
xAxis 0 to (dataset.length-1) column index of the dataset for X-axis
width px width of the chart
height px height of the chart
chartType area Type of the chart
interpolationMode basis,step-after,step-before,linear,cardinal method of interpolation

Sample code for above charts


/******************************************************* Installation of the library ******************************************************/

             <link rel="stylesheet" href="bootstrap.min.css">
             <link rel="stylesheet" href="igviz.css">
             <script src="d3.min.js"> </script>
             <script src="vega.js"> </script>
             <script src="igviz.js"> </script>


/***********************************************************Invoking the Library **********************************************************/
             <script>

                    //dataset used to plot charts
                    var dataTable = {
                        "metadata" : {
                            "names" : ["Year","Sales","Expenses"],
                            "types" : ["T","N","N"]
                        },
                        "data" : [
                            ["2004",1000,400],
                            ["2005",1170, 460],
                            ["2006",660, 1120],
                            ["2007",1030,540]
                        ]
                    };

                    var width = document.getElementById("area").offsetWidth; //canvas width
                        var height = 300;   //canvas height

                        var config = {
                            "title": "Sample Multi Area Chart",
                            "yAxis": [1,2],
                            "xAxis": 0,
                            "width": width,
                            "height": height,
                            "chartType": "area"
                       }




                        //draw a bar chart on div #bar
                    chart=igviz.setUp("#bar",config,dataTable);
                    chart.plot(dataTable.data)

                    //responsive handler for window resize event
                    window.onresize=function(){chart.resize()};
            </script>
