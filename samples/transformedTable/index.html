Configurations

Property Value Description
title String Title of the table
xAxis 0 to (dataset.length-1) column index for comparison
width px width of the chart
height px height of the chart
chartType table Type of the chart
rowIndex 0 to (dataset.length-1) index that should be used as rows in transformed table
columnIndex 0 to (dataset.length-1) index that should be used as columns in transformed table
cellIndex 0 to (dataset.length-1) index that should be used as cells in transformed table
aggregate min|max|count|avg|sum aggregate function on cell values
fontBasedStyle true | false Font Size is varying with the numerical value
colorBasedStyle true | false Color is varying with the numerical value

Sample code for above charts


/******************************************************* Installation of the library ******************************************************/

             <link rel="stylesheet" href="bootstrap.min.css">
             <link rel="stylesheet" href="igviz.css">
             <script src="d3.min.js"> </script>
             <script src="igviz.js"> </script>


/********************************************************** Invoking the Library **********************************************************/
             <script>

                //dataset used to plot charts
                dataTable={

                    "metadata":{
                    "names":["Candidate","city_id","votes"],
                            "types":['C', 'N', 'N']
                },
                    "data": [
                    ['A',1,4],
                    ['A',2,0],
                    ['A',3,5],
                    ['A',5,3],
                    ['B',1,3],
                    ['B',2,5],
                    ['B',3,8],
                    ['B',5,9],
                    ['C',1,1],
                    ['C',2,9],
                    ['C',3,5],
                    ['C',5,6],
                    ['D',5,6],

                    ['A',1,7],
                ]


                //        igviz.Transform(sample,0,1,"sum",2);
                }
                var width = document.getElementById("colorBased").offsetWidth; //canvas width
                var height = 270;   //canvas height




                var colorBased2D = {
                    "title": "Area By Country",
                    "width": width,
                    "height": height,
                    "rowIndex":0,
                    "columnIndex":1,
                    "cellIndex":2,
                    "aggregate":"sum",
                    "chartType": "table",
                    "heatMap":true
                }



                igviz.draw("#colorBased",colorBased2D,dataTable);

            </script>
