Configurations

Property Value Description
title String Title of the table
yAxis [0-(dataset.length-1)] An array of indices of dataset for Y-axis
xAxis 0-(dataset.length-1) column index of the dataset for X-axis
width px width of the chart
height px height of the chart
chartType line Type of the chart
interpolationMode basis,step-after,step-before,linear,cardinal method of interpolation

Sample code for above charts


/******************************************************* Installation of the library ******************************************************/

             <link rel="stylesheet" href="igviz.css">
             <script src="d3.min.js"> </script>
             <script src="vega.js"> </script>
             <script src="igviz.js"> </script>


/********************************************************** Invoking the Library **********************************************************/
             <script>

                //dataset used to plot charts
                 var dataTable = {
                    "metadata" : {
                        "names" : ["Year","Sales","Expenses"],
                        "types" : ["T","N","N"]
                    },
                    "data" : [
                        ["2004",1000,400],
                        ["2005",1170, 460],
                        ["2006",660, 1120],
                        ["2007",1030,540]
                    ]
                };

                var width = document.getElementById("line").offsetWidth ; //canvas width
                var height = 270;   //canvas height

                var lineChartConfig = {
                       "xAxis": 0,
                       "yAxis": [1,2],
                       "width": width,
                       "height": height,
                       "textAngle" : -90,
                       "chartType": "line"
                   }



                //draw a line chart on div #line
                staticChart=igviz.setUp("#staticLine", lineChartConfig, dataTable);
                staticChart.setXAxis({zero:false});
                staticChart.plot(dataTable.data);


                //draw a dynamic line chart on div #bar
                chart=igviz.setUp("#dynamicLine", lineChartConfig, dataTable);
                chart.setXAxis({zero:false});
                chart.plot(dataTable.data);
                var indexX=6
                setInterval(function(){

                    number=Math.floor(Math.random()*1000)
                    number2=Math.floor(Math.random()*100)
                    console.log(number2,number*number2)
                    chart.update([indexX,number,number2]);
                    indexX++;
                },2000)


                //resizeHandler
                window.onresize=function(){chart.resize()};

            </script>
