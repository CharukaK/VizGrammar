Configurations

Property Value Description
title String Title of the table
yAxis 0 to (dataset.length-1) column index of the dataset for Y-axis
xAxis 0 to (dataset.length-1) column index of the dataset for X-axis
pointLabel 0 to (dataset.length-1) column index of the dataset for text labels
width px width of the chart
height px height of the chart
chartType bar Type of the chart
textAngle (-360,+360) Angle between lables and X-axis
barColor color code color of the bars

Sample code for above charts


/******************************************************* Installation of the library ******************************************************/

             <link rel="stylesheet" href="bootstrap.min.css">
             <link rel="stylesheet" href="igviz.css">
             <script src="d3.min.js"> </script>
             <script src="vega.js"> </script>
             <script src="igviz.js"> </script>


/***********************************************************Invoking the Library **********************************************************/
             <script>

                    //dataset used to plot charts
                        var dataTable = {
                          "metadata":{
                            "names":["Month","Person","Sales"],
                            "types":['C', 'C','N']
                          },
                          "data": [
                              ['January','A',5],
                              ['February','A',8],
                              ['March','A',3],
                              ['April','A',5],
                              ['May','A',9],
                              ['June','A',0],
                              ['July','A',3],
                              ['August','A',0],
                              ['September','A',1],
                              ['October','A',3],
                              ['November','A',9],
                              ['December','A',8],
                              ['January','B',2],
                              ['February','B',5],
                              ['March','B',2],
                              ['April','B',8],
                              ['May','B',7],
                              ['June','B',5],
                              ['July','B',5],
                              ['August','B',5],
                              ['September','B',0],
                              ['October','B',3],

                              ['December','B',4],
                              ['January','C',4],
                              ['February','C',4],
                              ['March','C',2],

                              ['May','C',9],
                              ['June','C',4],
                              ['July','C',9],
                              ['August','C',1],
                              ['September','C',6],
                              ['October','C',0],
                              ['November','C',6],
                              ['December','C',8],
                          ]
                        }

                        var width = document.getElementById("bar").offsetWidth; //canvas width
                        var height = 400;   //canvas height

                        var config = {
                            "title": "Sample Stacked Bar Chart",
                            "yAxis": 2,
                            "xAxis": 0,
                            "groupedBy":1,
                            "width": width,
                            "height": height,
                            "chartType": "bar",
                            "format":"stacked"
                       }




                        //draw a stacked bar chart on div #bar
                    chart=igviz.setUp("#bar",config,dataTable);
                    chart.plot(dataTable.data)

                    window.onresize=function(){chart.resize()};
            </script>
